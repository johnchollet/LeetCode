438. Find All Anagrams in a String

Given a string s and a non-empty string p, find all the start indices of p's anagrams in s.

Strings consists of lowercase English letters only and the length of both strings s and p will not be larger than 20,100.

The order of output does not matter.

Example 1:

Input:
s: "cbaebabacd" p: "abc"

Output:
[0, 6]

Explanation:
The substring with start index = 0 is "cba", which is an anagram of "abc".
The substring with start index = 6 is "bac", which is an anagram of "abc".

Example 2:

Input:
s: "abab" p: "ab"

Output:
[0, 1, 2]

Explanation:
The substring with start index = 0 is "ab", which is an anagram of "ab".
The substring with start index = 1 is "ba", which is an anagram of "ab".
The substring with start index = 2 is "ab", which is an anagram of "ab".
C++ 方法一：暴力法:
class Solution {
public:
vector<int> findAnagrams(string s, string p) {
if (s.empty()) return {};
vector<int> res, cnt(128, 0);
int ns = s.size(), np = p.size(), i = 0;
for (char c : p) ++cnt[c];
while (i < ns) {
bool success = true;
vector<int> tmp = cnt;
for (int j = i; j < i + np; ++j) {
if (--tmp[s[j]] < 0) {
success = false;
break;
}
}
if (success) {
res.push_back(i);
}
++i;
}
return res;
}
};
假如 s 的长度为 n，p 的长度为 m。时间复杂度: O(nm)
解法一为什么慢？每次从 s 第 i 个位置往后取 p 个长度去比较，比完 i + 1。
c++ 方法二：维护一个滑窗
class Solution {
public:
vector<int> findAnagrams(string s, string p) {
int ns = s.size(), np = p.size();
if (ns < np) return {};
vector<int> res;
vector<int> ms(26, 0);
vector<int> mp(26, 0);
for(int i = 0; i < np; i++){
mp[p[i]-'a'] += 1;
ms[s[i]-'a'] += 1;
}
if (ms == mp) res.push_back(0);
int i = 1;
while(i < ns - np + 1){
ms[s[i+np-1] - 'a'] += 1;
ms[s[i-1] - 'a'] -= 1;
if (ms == mp){
res.push_back(i);
}
i++;
}
return res;
}
};
上面的解法维护一个数组 ms, 统计一个长为 s 上一个 p 长度的小滑窗。如果 ms == mp (无序，只要分布相同即可)，那么就匹配上了。ms 往后移动一格时，只需要把新来的右侧字符和旧的最左侧字符统计更新一下，再比较 ms == mp 即可。

但其实这样虽然维护了滑窗，但还是不够快，为什么？ms == mp 这个操作挺费时的。

优化 ms == mp 这个比较操作就可以优化到极限了。

因此有了解法三：抛弃 ms == mp 比较操作，用一个 cnt 变量指示匹配情况，ms 往后移动一格时，只需要根据新来的右侧字符和旧的最左侧字符更新一下 cnt, 如果 cnt == 0 了，那就匹配成功了：
class Solution {
public:
vector<int> findAnagrams(string s, string p) {
int ns = s.size(), np = p.size();
if (ns < np) return {};
vector<int> res, ms(26, 0);
// 维护一个统计数组 ms
for(int i = 0; i < np; i++){
ms[p[i]-'a'] += 1;
}
int left = 0, right = 0, cnt = np;
while(right < ns){
if (ms[s[right]-'a'] > 0) cnt--; // 匹配了，cnt--
// 更新剩余匹配次数，这时不在 p 中的会变成负数
ms[s[right]-'a']--;
right++;
if (cnt == 0) res.push_back(left);
if (right - left == np){
if (ms[s[left]-'a'] >= 0) cnt++;
ms[s[left]-'a']++;
left++;
}
}
return res;
}
};
